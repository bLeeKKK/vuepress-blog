# 构造签名

使用 new 调用函数 
```ts
class Cort {
  name: string
  constructor(s: string) {
    this.name = s
  }
}

type TCort = {
  new(name: string): Cort
}

function doSom(fn: TCort) {
  return new fn("我我我")
}

let m = doSom(Cort)

console.log(m)
```

同时定义构造签名、调用签名
```ts
interface TDate {
  new(s: string): Date
  (n: number): number
}

function doSom(fn: TDate) {
  let a = new fn("2022-4-8")
  let b = fn(22)
  console.log(a, b)
}
```
